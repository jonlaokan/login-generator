4   import java.lang.StringBuilder;

6   public class PasswordGeneration {
7   
8       static final int LEN = 8;
9       static SecureRandom rnd = new SecureRandom();
10      static final String alphanum =
11              "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz";
12  
13  
14      /**
15       * Create a Random String which may contain :
16       *  - numbers : 0 to 9
17       *  - lowercase letters : a-z
18       *  - uppercase letters : A-Z
19       *
20       * @return a random LEN character String
21       */
22      public String getRandomPassword() {
23          StringBuilder sb = new StringBuilder(LEN);
24  
25          for (int i = 0; i < LEN; i++)
26              sb.append(alphanum.charAt(rnd.nextInt(alphanum.length())));
27  
28          return sb.toString();
29      }
